2.進位制數轉換
  十進位制轉其他進位制
        print(bin(100))  # 0b1100100
        print(oct(100))  # 0o144
        print(hex(100))  # 0x64
  '''
  數字的開頭如果是0b則為二進位制 0o則為八進位制 0x則為十六進位制
  '''
  其他進位制轉十進位制
        print(int(0b1100100))
        print(int(0o144))
        print(int(0x64))
      
        print(int("0b1100100", 2))
        print(int("0o144", 8))
        print(int("0x64", 16))

                    print('使用者名稱已存在')
                else:
                    # 8.列表添加使用者名稱
                    data_list.append(username)
                    print(f'使用者名稱{username}添加成功')
            elif choice_num == '2':
                # 9.循環列印每一個用戶數據
                for name in data_list:
                    print(
                    f"""
                    ------------user info---------
                    使用者名稱:{name}
                    ------------------------------
                    """)
            elif choice_num == '3':
                # 10.獲取用戶想要刪除的使用者名稱
                delete_username = input('請輸入您想要刪除的使用者名稱>>>:').strip()
                # 11.先判斷使用者名稱是否存在
                if delete_username in data_list:
                    data_list.remove(delete_username)
                    print(f'使用者名稱{delete_username}刪除成功')
                else:
                    print('使用者名稱不存在!!!')
            else:
                print('很抱歉 暫時沒有您想要執行的功能編號')
